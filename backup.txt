app.get('/auth/login', (req, res) => {
  const clientId = process.env.CLIENT_ID;
  const redirect = encodeURIComponent(process.env.REDIRECT_URI);
 try {
  const scopes = encodeURIComponent([
    'https://api.ebay.com/oauth/api_scope',
    'https://api.ebay.com/oauth/api_scope/sell.inventory',
    'https://api.ebay.com/oauth/api_scope/sell.account',
    'https://api.ebay.com/oauth/api_scope/sell.account.readonly',
    'https://api.ebay.com/oauth/api_scope/sell.fulfillment'
  ].join(' '));

  const authUrl =
    `https://auth.ebay.com/oauth2/authorize` +
    `?client_id=${clientId}` +
    `&response_type=code` +
    `&redirect_uri=${redirect}` +
    `&scope=${scopes}`;

  res.redirect(authUrl);
 } catch (error) {
  const errorMessage = error.response ? error.response.data : error.message;
  return res.status(500).json({ error: 'Authorization URL generation failed', details: errorMessage });
 }
});

app.get('/auth/success', async (req, res) => {
  const { code } = req.query;
  if (!code) {
    return res.status(400).send('Missing authorization code');
  }

  const creds = Buffer.from(
    `${process.env.CLIENT_ID}:${process.env.CLIENT_SECRET}`
  ).toString('base64');

  const body = qs.stringify({
    grant_type: 'authorization_code',
    code: code,
    redirect_uri: process.env.REDIRECT_URI,
  });

  try {
    const { data } = await axios.post(
      `https://api.ebay.com/identity/v1/oauth2/token`,
      body,
      {
        headers: {
          Authorization: `Basic ${creds}`,
          'Content-Type': 'application/x-www-form-urlencoded',
        },
      }
    );

    const { access_token, refresh_token, expires_in } = data;

    // Store tokens properly
    user.ebay.accessToken = access_token;
    user.ebay.refreshToken = refresh_token;
    
    // Calculate and store expiry time
    const expiryTime = new Date(Date.now() + expires_in * 1000);
    user.ebay.accessToken_EXPIRY = expiryTime.toISOString();

    // Redirect to a success page or return JSON
    return res.json({ 
      success: true,
      message: "Authorization successful! You can now use the eBay APIs.",
      auth_token: access_token,
      expires_at: expiryTime,
      expires_in_seconds: expires_in
    });
  } catch (err) {
    console.error('Token exchange failed:', err.response?.data || err);
    return res.status(500).send('Failed to exchange authorization code');
  }
});

// â”€â”€ 4) Get Current Auth Token API â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
app.get('/auth/token', async (req, res) => {
  try {
    // Check if we have a token in memory/environment
    if (!user.ebay.accessToken || !user.ebay.accessToken_EXPIRY) {
      // No token exists, try to refresh it
      return refreshAndReturnToken(req, res);
    }
    
    // Check if token is expired (with 5-minute buffer)
    const tokenExpiry = new Date(user.ebay.accessToken_EXPIRY);
    const now = new Date();
    const fiveMinutesFromNow = new Date(now.getTime() + 5 * 60 * 1000);
    
    if (tokenExpiry < fiveMinutesFromNow) {
      // Token is expired or will expire soon, refresh it
      return refreshAndReturnToken(req, res);
    }
    
    // Return the current valid token
    return res.json({
      success: true,
      auth_token: user.ebay.accessToken,
      expires_at: tokenExpiry,
      expires_in_seconds: Math.floor((tokenExpiry - now) / 1000)
    });
  } catch (error) {
    console.error('Error retrieving auth token:', error);
    return res.status(500).json({
      success: false,
      message: 'Failed to retrieve auth token',
      error: error.message
    });
  }
});

// Helper function to refresh and return token
async function refreshAndReturnToken(req, res) {
  if (!user.ebay.refreshToken) {
    return res.status(400).json({ 
      success: false,
      message: 'No refresh token configured. Please authenticate first.'
    });
  }
  const creds = Buffer.from(
    `${process.env.CLIENT_ID}:${process.env.CLIENT_SECRET}`
  ).toString('base64');
  
  const body = qs.stringify({
    grant_type: 'refresh_token',
    refresh_token: user.ebay.refreshToken,
    scope: [
      'https://api.ebay.com/oauth/api_scope',
      'https://api.ebay.com/oauth/api_scope/sell.inventory',
      'https://api.ebay.com/oauth/api_scope/sell.account',
      'https://api.ebay.com/oauth/api_scope/sell.account.readonly',
      'https://api.ebay.com/oauth/api_scope/sell.fulfillment'
    ].join(' '),
  });

  console.log('Refreshing auth token...', `${user.ebay.refreshToken}:${process.env.CLIENT_SECRET}`);

  try {
    const { data } = await axios.post(
      `https://api.ebay.com/identity/v1/oauth2/token`,
      body,
      {
        headers: {
          Authorization: `Basic ${creds}`,
          'Content-Type': 'application/x-www-form-urlencoded',
        },
      }
    );

    const { access_token, refresh_token: newRt, expires_in } = data;

    // Save the new tokens
    user.ebay.accessToken = access_token;
    
    // Calculate and store expiry time
    const expiryTime = new Date(Date.now() + expires_in * 1000);
    user.ebay.accessToken_EXPIRY = expiryTime.toISOString();
    
    // Save the new refresh token if provided
    if (newRt) {
      user.ebay.refreshToken = newRt;
    }

    return res.json({ 
      success: true,
      auth_token: access_token,
      expires_at: expiryTime,
      expires_in_seconds: expires_in
    });
  } catch (err) {
    console.error('Token refresh failed:', err.response?.data || err);
    return res.status(500).json({
      success: false,
      message: 'Failed to refresh auth token',
      error: err.response?.data || err.message
    });
  }
}
